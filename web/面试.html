  1. 实现 randomRangNum 方法，接受 max,min,n 三个参数,返回一个数组 要求 0
  • 数组长度为 n
  • 数组元素唯一
  • 数组元素的取值范围是 min~max 之间的正整数
js
function randomRangNum(max, min, n) {
  const arr = []
  const diff = max - min + 1
  if (n > diff ) throw new Error('n元素过大')
  while(n) {
    const el = Math.floor(Math.random() * min + diff)
    if (!arr.includes(el)){
      arr.push(el)
      n--
    }
  }

  return arr
}
